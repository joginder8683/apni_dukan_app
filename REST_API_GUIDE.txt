			}	
			
add product API

	POST	http://host:port/add_product/
	
	argument:
		add product without any discount code
					{
						"name":"product_1",
						"price":1299,
						"desc":"first product added"
			
					}
					
		add product with any discount code
		
					{
						"name":"product_2",
						"price":1199,
						"desc":"second product added",
						"coupon":1	
					}
					


add product to cart

	POST	http://host:port/add_product_in_cart/
	argument:
		add product in new cart

	"coupon":1,
